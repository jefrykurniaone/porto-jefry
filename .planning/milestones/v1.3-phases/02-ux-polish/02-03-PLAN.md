  <action>
Create custom error boundary component at `src/app/[locale]/error.tsx` (per D-05, D-06).

**File structure per UI-SPEC.md layout:**

```tsx
'use client';

import { useEffect } from 'react';
import { useTranslations, useLocale } from 'next-intl';
import { Link } from '@/i18n/routing';
import { AlertTriangle } from 'lucide-react';

interface ErrorProps {
  error: Error & { digest?: string };
  reset: () => void;
}

export default function Error({ error, reset }: ErrorProps) {
  const t = useTranslations('error');
  const locale = useLocale();

  useEffect(() => {
    // Log error to console for debugging (not exposed to user)
    // eslint-disable-next-line no-console
    console.error('[Error Boundary]', error);
  }, [error]);

  return (
    <div className="min-h-screen flex items-center justify-center px-4">
      <div className="text-center max-w-md">
        <AlertTriangle 
          size={64} 
          className="mx-auto mb-6 text-yellow-500 dark:text-yellow-400" 
          aria-hidden="true"
        />
        <h1 className="text-4xl font-semibold mb-4 text-gray-900 dark:text-white">
          {t('title')}
        </h1>
        <p className="text-lg text-gray-600 dark:text-gray-300 mb-8">
          {t('message')}
        </p>
        <div className="flex gap-4 justify-center">
          <button
            type="button"
            onClick={reset}
            className="px-6 py-4 bg-blue-600 hover:bg-blue-700 text-white font-semibold rounded-xl transition-colors"
          >
            {t('tryAgain')}
          </button>
          <Link href={`/${locale}`}>
            <button
              type="button"
              className="px-6 py-4 border border-blue-600 text-blue-600 dark:text-blue-400 dark:border-blue-400 hover:bg-blue-50 dark:hover:bg-blue-950 font-semibold rounded-xl transition-colors"
            >
              {t('returnHome')}
            </button>
          </Link>
        </div>
      </div>
    </div>
  );
}
```

**Implementation notes:**
- **MUST use `'use client'` directive** (required per Next.js error boundary conventions — error boundaries must be client components)
- Import Link from `@/i18n/routing` (not next/navigation)
- ErrorProps interface defines props shape: `error` (Error object with optional digest) and `reset` (function to retry)
- AlertTriangle icon size 64px, yellow color (warning semantic color per UI spec)
- Icon has `aria-hidden="true"` (decorative)
- Spacing matches UI spec: mb-6 (icon-title), mb-4 (title-message), mb-8 (message-buttons)
- Two-button layout with gap-4 (16px) between buttons
- Primary button (Try Again): blue background, calls `reset` prop
- Secondary button (Return Home): outlined style, links to `/{locale}`
- Both buttons have `type="button"` attribute (per project conventions)
- Typography: text-4xl (36px) for title, text-lg (18px) for message
- Dark mode support via `dark:` Tailwind variants
- useEffect logs error to console for debugging (not exposed to user)

**File location:** `src/app/[locale]/error.tsx` (must be in locale directory per Next.js App Router conventions)

**Per D-07:** Error boundary must work for both EN and ID locales — verify by triggering runtime errors in both locale routes.
