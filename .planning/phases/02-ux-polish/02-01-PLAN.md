```typescript
it('shows error banner when CV download fails, then auto-dismisses', async () => {
  vi.useFakeTimers();
  
  // Mock fetch to fail with HTTP 500
  const fetchMock = vi.fn().mockResolvedValue({ 
    ok: false, 
    status: 500,
    blob: () => Promise.reject(new Error('HTTP 500'))
  });
  vi.stubGlobal('fetch', fetchMock);

  renderHero();
  const user = userEvent.setup({ delay: null }); // Disable delay for fake timers
  
  // Click CV download button
  const buttons = screen.getAllByRole('button');
  const cvButton = buttons.find(b => /cv|unduh/i.test(b.textContent ?? ''));
  await user.click(cvButton!);

  // Wait for error state update
  await waitFor(() => {
    const banner = screen.queryByRole('alert');
    expect(banner).toBeInTheDocument();
    expect(banner).toHaveTextContent(/failed.*generate.*cv.*http.*500/i);
    expect(banner).toHaveAttribute('aria-live', 'polite');
  });

  // Verify AlertCircle icon is present
  const icon = screen.getByRole('alert').querySelector('svg');
  expect(icon).toBeInTheDocument();

  // Fast-forward 5 seconds
  vi.advanceTimersByTime(5000);

  // Banner should be removed
  await waitFor(() => {
    expect(screen.queryByRole('alert')).not.toBeInTheDocument();
  });

  vi.useRealTimers();
  vi.unstubAllGlobals();
});
```

3. Error text includes "HTTP 500" (status code per D-02)

5. AlertCircle icon renders (per UI spec)

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information Disclosure | Error banner | mitigate | Display generic "Failed to generate CV (HTTP {status})" — no stack traces or internal error details exposed to user (error details only in console for debugging) |
| T-02-02 | Denial of Service | Auto-dismiss timer | accept | 5-second setTimeout is client-side only; no server resource impact; multiple rapid clicks are already rate-limited by existing rate limiter at API level |
| T-02-03 | Tampering | Translation keys | accept | Translation keys are bundled at build time; tampering requires source code access (same trust level as any client-side code) |
